# TrendRadar Docker 环境变量示例
#
# 使用方法：
# 1) 复制本文件为 docker/.env
# 2) 修改 FEISHU_WEBHOOK_URL 等变量
# 3) 启动：cd docker && docker compose up -d trendradar
#
# 安全提醒：docker/.env 通常在 .gitignore 中，请勿提交包含 webhook/token 的文件。

# ========== 定时运行（必配/常用） ==========
# cron 表达式（5 段）：分 时 日 月 周
# 默认：每 30 分钟执行一次
CRON_SCHEDULE=*/30 * * * *

# 运行模式：
# - cron：定时执行（推荐）
# - once：只执行一次就退出
RUN_MODE=cron

# 容器启动后是否先立刻执行一次（推荐 true，便于测试）
IMMEDIATE_RUN=true

# ========== 飞书推送（必配） ==========
# 飞书机器人 Webhook 地址（多账号用 ; 分隔）
FEISHU_WEBHOOK_URL=https://www.feishu.cn/flow/api/trigger-webhook/xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx

# ========== 可选：内置 Web 服务器（不需要可保持 false） ==========
ENABLE_WEBSERVER=false
WEBSERVER_PORT=8080

# ========== 可选：AI 分析（无 AI_KEY 建议关闭） ==========
AI_ANALYSIS_ENABLED=false
AI_API_KEY=
AI_MODEL=
AI_API_BASE=
